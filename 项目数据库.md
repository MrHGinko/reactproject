```js
// 精简化订单状态
{
    订单号;
    (关联)用户手机号; [] 
    (关联)司机手机号;
    人数;
    乘客信息; => 一定包含下单用户信息, 还需填写
    {
	    // 默认添加生成订单用户
        乘客姓名;
        乘客电话;
        乘客身份证号;
    }
    
    出发地;
    目的地;
    下单时间; 标记状态为待派车

    发车时间; 标记状态为已发车
    到达时间; 标记状态为已结束
    发车时备注; (用户)
    
    乘车方式=>[包车, 拼车]
    支付金额; 里程数*0.1
    支付方式;
    评价状态; (已结束后开放评价)
    订单状态 => [待派车, 已发车, 已结束, 退单(特殊)];
}

// 添加订单需求用户信息
// 关联 订单中添加乘客,
// 关联 会员统计数据
{
    姓名;
    身份证号;
    手机;
    消费金额;
    呼叫次数;
    评价; (查看相应订单的评价信息)
    {
        订单号; (根据展示内容通过订单号获取对应已结束订单的信息)
        订单评价内容;
        评分;
    }
    
    账号状态 => [正常, 禁用]
}

// 司机信息
// 关联 订单中添加司机
// 关联车辆统计.
{
    车牌号;
    车辆类型;
    座位数;
    司机姓名;
    司机电话;
    
    司机状态; [空闲, 忙碌]

    任务; => 关联订单中为[待派车, 已发车(若有, 只有1个)]的订单信息
    {
	订单号;
    }
    
    接单总金额; [已结束订单的金额计算]
}

用户下单 =>
	主要通过车辆座位数, 车辆类型, 司机电话, 司机姓名进行检索;
    填写支付方式, 出发地, 目的地, 相应乘客的信息(默认包括自身);
	(可对订单进行退单操作, 已结束订单进行评价操作)
     
     (金额 需要通过地址间进行计算)
 
司机 =>
	关联到自身订单, 可选择是否接受该订单.
	对关联到自身的订单进行编辑, 填写发车时间, 改变订单的状态(已发车)
	
	
    


```
